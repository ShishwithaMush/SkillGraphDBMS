package com.dbms;

import com.dbms.model.User; import com.dbms.ui.AuthView; import com.dbms.ui.DashboardView; import javax.swing.*; import java.awt.*;

public class MainApp { private static int loggedInUserId=-1; private static User loggedInUser; private final JFrame frame=new JFrame("Skill Graph - Placement Preparation System"); public MainApp(){frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);frame.setMinimumSize(new Dimension(1180,760));showLogin();frame.setLocationRelativeTo(null);frame.setVisible(true);} public static int getLoggedInUserId(){return loggedInUserId;} public static User getLoggedInUser(){return loggedInUser;} public void showLogin(){loggedInUserId=-1;loggedInUser=null;set(new AuthView(this).login());} public void showSignup(){set(new AuthView(this).signup());} public void login(User u){loggedInUser=u;loggedInUserId=u.getUserId();frame.setMinimumSize(new Dimension(1280,800));set(new DashboardView(this).create());} public void logout(){showLogin();} private void set(JPanel p){frame.setContentPane(p);frame.revalidate();frame.repaint();frame.setLocationRelativeTo(null);} public static void main(String[] args){SwingUtilities.invokeLater(MainApp::new);} }
