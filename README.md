package com.dbms.service;
import com.dbms.DBConnection; import com.dbms.model.MockTest; import java.sql.*; import java.time.LocalDate; import java.util.*;
public class MockTestService { public List<MockTest> fetchTests()throws SQLException{List<MockTest> out=new ArrayList<>();String sql="SELECT test_id,test_name,category,total_marks FROM MOCK_TESTS ORDER BY category,test_name";try(Connection c=DBConnection.getConnection(); PreparedStatement ps=c.prepareStatement(sql); ResultSet rs=ps.executeQuery()){while(rs.next())out.add(new MockTest(rs.getInt("test_id"),rs.getString("test_name"),rs.getString("category"),rs.getInt("total_marks")));}return out;} public void saveResult(int userId,int testId,int score)throws SQLException{try(Connection c=DBConnection.getConnection(); PreparedStatement ps=c.prepareStatement("INSERT INTO TEST_RESULTS(user_id,test_id,score,date_taken) VALUES(?,?,?,?)")){ps.setInt(1,userId);ps.setInt(2,testId);ps.setInt(3,score);ps.setDate(4,java.sql.Date.valueOf(LocalDate.now()));ps.executeUpdate();}} }
